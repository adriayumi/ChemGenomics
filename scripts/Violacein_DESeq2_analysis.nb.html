# ORDERING RESULTS BY PADJ AND CREATING DATAFRAME

resOrdered <- res[order(res$padj),]

df <- resOrdered %>%
  as.data.frame() %>%
  rownames_to_column("ORF") %>%
  arrange(abs(padj)) %>%
  mutate(significative=if_else(padj<=.05, "yes", "no")) %>%
  mutate(up=if_else(log2FoldChange<0, "DMSO", "drug"))
